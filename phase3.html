# Week 1 simulation — first and second-order step responses
import numpy as np
from scipy import signal
import matplotlib.pyplot as plt

# First-order: G(s) = K / (tau*s + 1)
K, tau = 5.0, 0.2
sys1 = signal.TransferFunction([K], [tau, 1])
t, y = signal.step(sys1)
print(f"Steady-state: {y[-1]:.3f} (should be {K})")
print(f"Time constant tau visible at y = {K*0.632:.3f}")

# Second-order: G(s) = wn^2 / (s^2 + 2*zeta*wn*s + wn^2)
wn = 10.0  # natural frequency rad/s
for zeta in [0.1, 0.5, 0.707, 1.0, 2.0]:
    sys2 = signal.TransferFunction([wn**2], [1, 2*zeta*wn, wn**2])
    t, y = signal.step(sys2)
    overshoot = (max(y) - 1.0) * 100
    print(f"zeta={zeta:.3f}: overshoot={overshoot:.1f}%")
    plt.plot(t, y, label=f'ζ={zeta}')
plt.legend(); plt.xlabel('Time (s)'); plt.title('Effect of damping ratio')
plt.show()

DC motor transfer function

Mechanical: J(dω/dt) = Kt·i − B·ω − TL

Three anti-windup strategies
Clamping: stop integrating when saturated and error has same sign as outputSimplest. Works well for most cases. Use this first.
Back-calculation: feed saturation difference back through gain KtSmoother recovery. One extra tuning parameter.
Integrator freeze: halt integrator entirely during saturation
Discrete PID — Python class (then translate verbatim to C)
class DiscretePID:
    def __init__(self, Kp, Ki, Kd, Ts, out_min, out_max, N=10.0):
        self.Kp, self.Ki, self.Kd = Kp, Ki * Ts, Kd / Ts
        self.N = N
        self.out_min, self.out_max = out_min, out_max
        self.integrator   = 0.0
        self.differentiator = 0.0
        self.prev_meas    = 0.0

    def update(self, setpoint, measurement):
        error = setpoint - measurement

        # Proportional
        P = self.Kp * error

        # Integral with clamping anti-windup
        I_candidate = self.integrator + self.Ki * error
        if (P + I_candidate) < self.out_max and (P + I_candidate) > self.out_min:
            self.integrator = I_candidate   # only integrate if not saturated

        # Derivative on MEASUREMENT — avoids derivative kick on setpoint step
        # First-order IIR filter: reduces high-frequency noise amplification
        self.differentiator = (self.N * self.Kd * (self.prev_meas - measurement)
                               + self.differentiator * (1.0 - self.N * (1.0/10.0)))
        self.prev_meas = measurement

        output = P + self.integrator + self.differentiator
        return max(self.out_min, min(self.out_max, output))

Tuning order — always follow this sequence
Step 1 — P only: Ki=0, Kd=0. Increase Kp until response is fast but with some overshoot (~20–30%). System stable.
Step 2 — Add I: increase Ki until steady-state error is eliminated. Watch for oscillation — reduce Ki if it appears.
Step 3 — Add D: increase Kd to damp overshoot. Kd is sensitive to noise — use the derivative filter (N parameter).
Never tune all three simultaneously. You will not find the problem when something goes wrong.

C PID implementation
typedef struct {
    float Kp, Ki, Kd, Ts, N;
    float out_min, out_max;
    float integrator;
    float differentiator;
    float prev_measurement;
} PID_t;

void PID_Init(PID_t *p, float Kp, float Ki, float Kd,
              float Ts, float N, float lo, float hi) {
    p->Kp = Kp; p->Ki = Ki * Ts; p->Kd = Kd / Ts;
    p->N = N; p->Ts = Ts; p->out_min = lo; p->out_max = hi;
    p->integrator = p->differentiator = p->prev_measurement = 0.0f;
}

float PID_Update(PID_t *p, float setpoint, float measurement) {
    float error = setpoint - measurement;
    float P = p->Kp * error;

    /* Integral with clamping anti-windup */
    float I_new = p->integrator + p->Ki * error;
    float out_unsat = P + I_new;
    if (out_unsat >= p->out_min && out_unsat <= p->out_max)
        p->integrator = I_new;

    /* Derivative on measurement — no kick, filtered */
    p->differentiator = p->N * p->Kd * (p->prev_measurement - measurement)
                      + p->differentiator * (1.0f - p->N * p->Ts);
    p->prev_measurement = measurement;

    float out = P + p->integrator + p->differentiator;
    if (out > p->out_max) out = p->out_max;
    if (out < p->out_min) out = p->out_min;
    return out;
}

Velocity estimation — two methods
Fixed-period: count encoder pulses in fixed time window. v = Δcounts × (2π/PPR) / TsGood at high speed. Poor resolution at low speed (few pulses per window).
Fixed-pulse: measure time between encoder pulses using input capture. v = (2π/PPR) / ΔtGood at low speed. Poor at high speed (timer overflows). Combine both: switch based on measured speed.

Stability margins — practical meaning
Gain margin: how much you can increase Kp before instability. Target: GM > 6 dB.
Phase margin: how much phase lag you can add before instability. Target: PM = 45–60°.Less than 30°: oscillatory. More than 80°: sluggish. 45–60° is the engineering sweet spot.
Increasing Kp reduces phase margin. Derivative adds phase (good). Integral subtracts phase at high frequencies (bad).

IIR velocity filter — implement in C alongside PID
/* First-order IIR (exponential moving average)
   alpha = Ts / (Ts + 1/(2*pi*fc))
   fc  = cutoff frequency in Hz — typically bandwidth/5 to bandwidth/10
   Ts  = sample period
   Rule: too low alpha = smooth but laggy. Too high = noisy but responsive. */

#define FILTER_ALPHA   0.15f

static float s_vel_filtered = 0.0f;

/* Called every control loop tick, before PID_Update() */
float Filter_Velocity(float raw) {
    s_vel_filtered = FILTER_ALPHA * raw
                   + (1.0f - FILTER_ALPHA) * s_vel_filtered;
    return s_vel_filtered;
}

/* Hall sensor state (3-bit) → which phases are HIGH / LOW / floating
   Read Hall A, B, C from GPIO. Combine into a 3-bit index.
   Verify rotation direction — swap two phase wires if reversed.      */

typedef struct { uint8_t hi; uint8_t lo; uint8_t off; } CommStep_t;

/* Phases: 0=U, 1=V, 2=W */
static const CommStep_t COMM_FWD[8] = {
    {0,0,0},        /* 0b000 — invalid */
    {2, 1, 0},      /* Hall=1: W+  V-  U float */
    {0, 2, 1},      /* Hall=2: U+  W-  V float */
    {0, 1, 2},      /* Hall=3: U+  V-  W float */
    {1, 0, 2},      /* Hall=4: V+  U-  W float */
    {2, 0, 1},      /* Hall=5: W+  U-  V float */
    {1, 2, 0},      /* Hall=6: V+  W-  U float */
    {0,0,0},        /* 0b111 — invalid */
};

/* Called from Hall sensor EXTI interrupt */
void Hall_EXTI_Handler(void) {
    uint8_t hall = (GPIOA->IDR >> HALL_PIN_OFFSET) & 0x7u;
    if (hall == 0 || hall == 7) { Enter_Fault(FAULT_ENCODER); return; }
    CommStep_t s = COMM_FWD[hall];
    Apply_Commutation(s.hi, s.lo, s.off, g_duty);
}

void Apply_Commutation(uint8_t hi, uint8_t lo, uint8_t off, uint16_t duty) {
    /* Set PWM duty on high-side, enable low-side, tri-state floating phase */
    for (uint8_t ph = 0; ph < 3; ph++) {
        if (ph == hi)       { Phase_SetPWM(ph, duty); }
        else if (ph == lo)  { Phase_SetLow(ph); }
        else                { Phase_SetHiZ(ph); }
    }
}

Week 7 — FOC fundamentals: Clarke, Park, and simulation

FOC Theory
reference framesClarke transform Park transformd-q axis Id vs IqInverse ParkSVPWM
The FOC idea — before the math
3-phase currents in a running motor are sinusoidal and rotating. Impossible to PID-control directly — the reference keeps changing.
FOC rotates a coordinate frame to follow the rotor. In this rotating frame, currents appear as DC values.
Now two independent PID controllers work on DC quantities: Id (flux, set to 0) and Iq (torque, commanded by speed loop).
Transform chain — memorise this
1. Measure Ia, Ib → compute Ic = −(Ia+Ib)
2. Clarke: (Ia, Ib, Ic) → (Iα, Iβ) — stationary 2D
3. Park: (Iα, Iβ, θe) → (Id, Iq) — rotating, aligned to rotor
4. Two PIDs: Id→Vd, Iq→Vq
5. Inv. Park: (Vd, Vq, θe) → (Vα, Vβ)
6. SVPWM: (Vα, Vβ) → PWM duty cycles for 3 phases
Clarke and Park — Python implementation and verification
import numpy as np, matplotlib.pyplot as plt

def clarke(Ia, Ib, Ic):
    """3-phase → 2-phase stationary (alpha-beta)"""
    Ialpha = Ia
    Ibeta  = (Ia + 2*Ib) / np.sqrt(3)
    return Ialpha, Ibeta

def park(Ialpha, Ibeta, theta):
    """Stationary → rotating (d-q)"""
    Id =  Ialpha * np.cos(theta) + Ibeta * np.sin(theta)
    Iq = -Ialpha * np.sin(theta) + Ibeta * np.cos(theta)
    return Id, Iq

def inv_park(Vd, Vq, theta):
    Valpha = Vd * np.cos(theta) - Vq * np.sin(theta)
    Vbeta  = Vd * np.sin(theta) + Vq * np.cos(theta)
    return Valpha, Vbeta

Space vector PWM — simplified implementation
def svpwm(Valpha, Vbeta, Vdc):
    """Alpha-beta voltages → 3-phase duty cycles [0, 1]"""
    Va =  Valpha
    Vb = -0.5*Valpha + (np.sqrt(3)/2)*Vbeta
    Vc = -0.5*Valpha - (np.sqrt(3)/2)*Vbeta
    # Zero-sequence injection for maximum DC bus utilisation
    V_min = np.minimum(np.minimum(Va, Vb), Vc)
    V_max = np.maximum(np.maximum(Va, Vb), Vc)
    V_off = (V_max + V_min) / 2
    Ta = (Va - V_off) / Vdc + 0.5
    Tb = (Vb - V_off) / Vdc + 0.5
    Tc = (Vc - V_off) / Vdc + 0.5
    return np.clip(Ta, 0, 1), np.clip(Tb, 0, 1), np.clip(Tc, 0, 1)

Q15 Clarke and Park in C
/* Q15: range [-1.0, +1.0) mapped to [-32768, 32767]
   Intermediate products use int32_t to avoid overflow  */
typedef int16_t q15_t;
typedef int32_t q31_t;
#define Q15(x)        ((q15_t)((x) * 32767.0f))
#define Q15_MUL(a,b)  ((q15_t)(((q31_t)(a) * (b)) >> 15))

/* 256-entry sin/cos LUT — pre-computed at startup or in flash */
/* Index by electrical angle: angle_idx = (theta_elec * 256) / (2*pi) */
static const q15_t SIN_LUT[256] = { /* ... generated values ... */ };
static inline q15_t q15_sin(uint8_t idx) { return SIN_LUT[idx]; }
static inline q15_t q15_cos(uint8_t idx) { return SIN_LUT[(idx + 64) & 0xFF]; }

void Clarke_Q15(q15_t Ia, q15_t Ib, q15_t *Ialpha, q15_t *Ibeta) {
    *Ialpha = Ia;
    /* Ibeta = (Ia + 2*Ib) / sqrt(3) = 0.57735*Ia + 1.15470*Ib */
    *Ibeta = Q15_MUL(Q15(0.57735f), Ia) + Q15_MUL(Q15(1.15470f), Ib);
}

void Park_Q15(q15_t Ialpha, q15_t Ibeta, uint8_t theta_idx,
              q15_t *Id, q15_t *Iq) {
    q15_t c = q15_cos(theta_idx);
    q15_t s = q15_sin(theta_idx);
    *Id =  Q15_MUL(Ialpha, c) + Q15_MUL(Ibeta,  s);
    *Iq = -Q15_MUL(Ialpha, s) + Q15_MUL(Ibeta,  c);
}

/* FOC ISR — runs at 20kHz, triggered by ADC (= PWM centre) */
void ADC1_2_IRQHandler(void) {
    /* Total budget: <40µs at 20kHz (50µs period) */
    q15_t Ia = ADC_ReadPhaseA();
    q15_t Ib = ADC_ReadPhaseB();
    /* Ic = -(Ia+Ib) by KCL — no third sensor needed */

    q15_t Ialpha, Ibeta, Id, Iq;
    Clarke_Q15(Ia, Ib, &Ialpha, &Ibeta);

    uint8_t theta_idx = Encoder_GetElecAngle256();  /* electrical angle */
    Park_Q15(Ialpha, Ibeta, theta_idx, &Id, &Iq);

    float Vd = PID_Update(&g_id_pid, 0.0f, Id / 32767.0f);  /* Id ref = 0 */
    float Vq = PID_Update(&g_iq_pid, g_Iq_ref, Iq / 32767.0f);

    float Valpha, Vbeta;
    InvPark(Vd, Vq, theta_idx, &Valpha, &Vbeta);
    SVPWM_Update(Valpha, Vbeta, VDC);

Cascade structure
Inner loop: Id and Iq PIDs at 20kHz. Bandwidth 1–5kHz.
Outer loop: speed PID at 1kHz. Output = Iq setpoint. Bandwidth 50–200Hz.
Rule: inner loop must be ≥5× faster than outer loop.If inner is too slow, outer loop sees it as a lag and becomes unstable.
Tuning order
1. Tune inner current loops first. Outer loop disabled. Step Iq directly.Target: <1ms current step, <5% overshoot
2. Enable outer speed loop only after inner is stable. P only first.Target: <50ms speed step, <10% overshoot
3. Add ramp generator before speed setpoint. Limits acceleration rate to protect motor from current spikes.
Ramp generator and FreeRTOS task structure
/* Ramp generator — limits rate of change of speed setpoint */
typedef struct { float output; float max_rate; float Ts; } Ramp_t;

float Ramp_Update(Ramp_t *r, float target) {
    float max_step = r->max_rate * r->Ts;
    float err = target - r->output;
    if      (err >  max_step) r->output += max_step;
    else if (err < -max_step) r->output -= max_step;
    else                      r->output  = target;
    return r->output;
}

/* FreeRTOS task hierarchy */
/* ADC ISR (not a task): 20kHz — Clarke→Park→Id/Iq PID→InvPark→SVPWM
   Posts speed measurement to queue every 20 cycles (1kHz)          */

void SpeedControl_Task(void *pv) {        /* Priority 6 (highest task) */
    float speed;
    while (1) {
        xQueueReceive(g_speed_q, &speed, portMAX_DELAY);
        float ramped = Ramp_Update(&g_ramp, g_speed_setpoint);
        float Iq_ref = PID_Update(&g_speed_pid, ramped, speed);
        taskENTER_CRITICAL();
        g_Iq_ref = Iq_ref;    /* atomic write — ISR reads this */
        taskEXIT_CRITICAL();
    }
}

3-layer fault protection hierarchy
Layer 1 — Hardware (fastest): analog comparator output connected to gate driver nFAULT/ENABLE pin. Fires in nanoseconds. No software involved. Must be present.DRV8323: configure OCP threshold via SPI. Hardware disables gate drives immediately on overcurrent.
Layer 2 — ISR (fast): ADC current check every 50µs (20kHz). If current exceeds software limit: TIM1->BDTR &= ~TIM_BDTR_MOE (immediate PWM disable).Catches sustained overcurrent that hardware threshold allows. Also detects phase current imbalance.
Layer 3 — Task (slow, comprehensive): FaultMonitor_Task at 1kHz. Checks voltages, temperature, encoder validity, watchdog heartbeats, CAN timeout.Handles thermal derating, undervoltage lockout, encoder fault, communication watchdog.
Fault state machine and thermal derating
typedef enum {
    MSTATE_INIT, MSTATE_READY, MSTATE_RUNNING,
    MSTATE_FAULT, MSTATE_FAULT_LATCH
} MotorState_t;

typedef enum {
    FAULT_NONE         = 0,
    FAULT_OVERCURRENT  = (1u << 0),
    FAULT_OVERVOLTAGE  = (1u << 1),
    FAULT_UNDERVOLTAGE = (1u << 2),
    FAULT_OVERTEMP     = (1u << 3),
    FAULT_ENCODER      = (1u << 4),
    FAULT_WATCHDOG     = (1u << 5),
} FaultCode_t;

void Enter_Fault(FaultCode_t code) {
    /* Step 1: disable PWM immediately (atomic — no IRQ needed) */
    __disable_irq();
    TIM1->BDTR &= ~TIM_BDTR_MOE;
    __enable_irq();

    g_fault_code   |= code;
    g_motor_state   = MSTATE_FAULT_LATCH;

    /* Step 2: log to backup SRAM (survives reset) */
    BKPSRAM->fault_code  = g_fault_code;
    BKPSRAM->fault_speed = g_speed;
    BKPSRAM->fault_Iq    = g_Iq_ref;
    BKPSRAM->valid       = 0xDEADBEEF;

    /* Step 3: report over CAN */
    CAN_Send_FaultFrame(g_fault_code);
}

/* Thermal derating — reduce max Iq as temperature rises */
float Thermal_GetMaxIq(float temp_C) {
    const float T_NOMINAL = 60.0f;
    const float T_MAX     = 100.0f;
    const float IQ_MAX    = 10.0f;
    if (temp_C <= T_NOMINAL) return IQ_MAX;
    if (temp_C >= T_MAX)     return 0.0f;
    return IQ_MAX * (1.0f - (temp_C - T_NOMINAL) / (T_MAX - T_NOMINAL));
}
/* Apply in inner loop: Iq_clamped = fminf(Iq_ref, Thermal_GetMaxIq(temp)) */
